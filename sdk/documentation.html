üíª Basic Usage

Connecting to the Network

#include "usdtgverse.h" int main() { // Initialize USDTgVerse client USDTgVerseClient* client = usdtgverse_client_new(); // Connect to mainnet if (usdtgverse_client_connect(client, "mainnet") != 0) { printf("Failed to connect to USDTgVerse network\n"); return 1; } printf("Connected to USDTgVerse network\n"); // Clean up usdtgverse_client_free(client); return 0; }
